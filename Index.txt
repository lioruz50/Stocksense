/* ── STATE ── */
let chartInstance = null;

/* ── PROXY CHAIN (ordered by reliability) ── */
const PROXIES = [
  url => `https://corsproxy.io/?${encodeURIComponent(url)}`,
  url => `https://api.allorigins.win/get?url=${encodeURIComponent(url)}`,
  url => `https://thingproxy.freeboard.io/fetch/${url}`,
];

async function proxyFetch(url, timeoutMs = 8000) {
  for (const proxyFn of PROXIES) {
    try {
      const controller = new AbortController();
      const tid = setTimeout(() => controller.abort(), timeoutMs);
      const resp = await fetch(proxyFn(url), { signal: controller.signal });
      clearTimeout(tid);
      if (!resp.ok) continue;
      const text = await resp.text();
      // allorigins wraps in {contents:...}, others return raw
      try {
        const parsed = JSON.parse(text);
        if (parsed.contents) return JSON.parse(parsed.contents);
        return parsed;
      } catch { return JSON.parse(text); }
    } catch (e) {
      console.warn('Proxy failed:', proxyFn(url).substring(0,60), e.message);
    }
  }
  throw new Error('כל ה-Proxy servers נכשלו. אנא נסה שוב בעוד דקה.');
}

/* ── HELPERS ── */
function fmt(n, digits = 2) {
  if (n == null || isNaN(n)) return 'N/A';
  if (Math.abs(n) >= 1e12) return (n/1e12).toFixed(2) + 'T';
  if (Math.abs(n) >= 1e9)  return (n/1e9 ).toFixed(2) + 'B';
  if (Math.abs(n) >= 1e6)  return (n/1e6 ).toFixed(2) + 'M';
  return n.toFixed(digits);
}
function fmtPct(n) { return n == null ? 'N/A' : (n*100).toFixed(1)+'%'; }
function fmtNum(n, d=2) { return n == null ? 'N/A' : Number(n).toFixed(d); }
function colorClass(val, goodThresh, badThresh, invert=false) {
  if (val == null) return '';
  const isGood = invert ? val < goodThresh : val > goodThresh;
  const isBad  = invert ? val > badThresh  : val < badThresh;
  return isGood ? 'good' : isBad ? 'bad' : 'warn';
}

/* ── QUICK SEARCH ── */
function qs(t) { document.getElementById('ticker-input').value = t; go(); }

/* ── UI TOGGLE ── */
function setState(state) {
  document.getElementById('loading-box').classList.toggle('on', state === 'loading');
  document.getElementById('result').classList.toggle('on', state === 'result');
  document.getElementById('error-box').classList.toggle('on', state === 'error');

}

function showError(msg) {
  document.getElementById('error-box').innerHTML = '⚠ ' + msg;
  setState('error');
}

/* ── MAIN ── */
async function go() {
  const raw = document.getElementById('ticker-input').value.trim().toUpperCase().replace(/[^A-Z.^]/g,'');
  if (!raw) return;
  currentTicker = raw;
  setState('loading');

  try {
    // Fetch both endpoints in parallel
    const [chartData, quoteData] = await Promise.all([
      proxyFetch(`https://query1.finance.yahoo.com/v8/finance/chart/${raw}?range=1mo&interval=1d&includePrePost=false`),
      proxyFetch(`https://query1.finance.yahoo.com/v10/finance/quoteSummary/${raw}?modules=financialData,defaultKeyStatistics,summaryDetail,assetProfile,incomeStatementHistory,earningsTrend`)
    ]);

    // Validate
    if (!chartData?.chart?.result?.[0]) throw new Error(`לא נמצאו נתוני מחיר עבור "${raw}". בדוק שהסימול נכון.`);
    if (!quoteData?.quoteSummary?.result?.[0]) throw new Error(`לא נמצאו נתונים פונדמנטליים עבור "${raw}".`);

    const priceResult = chartData.chart.result[0];
    const fundResult  = quoteData.quoteSummary.result[0];

    renderAll(priceResult, fundResult, raw);
    setState('result');

    // After render, stream AI analysis
    streamAIAnalysis(priceResult, fundResult, raw);

  } catch (e) {
    showError(e.message || 'שגיאה לא צפויה. נסה שוב.');
  }
}

/* ── RENDER MARKET DATA ── */

  const prev = meta.chartPreviousClose || meta.previousClose || 0;
  const cur  = meta.regularMarketPrice || 0;

  const chEl = document.getElementById('r-change');
  chEl.innerHTML = `<span style="color:${diff>=0?'var(--green)':'var(--red)'}">${diff>=0?'▲':'▼'} ${Math.abs(pct).toFixed(2)}% (${diff>=0?'+':''}${fmtNum(diff,2)})</span>`;

  // Chart
  renderChart(price, diff >= 0);

  // Metrics
  const pe    = stats.trailingPE?.raw;
  const fwdPE = stats.forwardPE?.raw;
  const pb    = stats.priceToBook?.raw;
  const ps    = stats.priceToSalesTrailing12Months?.raw;
  const mcap  = stats.marketCap?.raw;
  const beta  = stats.beta?.raw;
  const divYld= detail.dividendYield?.raw;
  const roe   = fin.returnOnEquity?.raw;
  const roa   = fin.returnOnAssets?.raw;
  const pm    = fin.profitMargins?.raw;
  const gm    = fin.grossMargins?.raw;
  const revGr = fin.revenueGrowth?.raw;
  const epsGr = stats.earningsQuarterlyGrowth?.raw;
  const d2e   = stats.debtToEquity?.raw;
  const cr    = fin.currentRatio?.raw;
  const fcf   = fin.freeCashflow?.raw;
  const targetP = fin.targetMeanPrice?.raw;

  const upside = (targetP && cur) ? ((targetP - cur)/cur*100) : null;

  const metrics = [
    { l:'P/E Trailing',    v: pe    != null ? fmtNum(pe,1)    : 'N/A', cls: colorClass(pe,    15, 35, true) },
    { l:'P/E Forward',     v: fwdPE != null ? fmtNum(fwdPE,1): 'N/A', cls: colorClass(fwdPE, 12, 30, true) },
    { l:'P/B (Price/Book)',v: pb    != null ? fmtNum(pb,2)   : 'N/A', cls: colorClass(pb,    1,  6,  true) },
    { l:'P/S (Price/Sales)',v:ps    != null ? fmtNum(ps,2)   : 'N/A', cls: '' },
    { l:'שווי שוק',        v: mcap  != null ? fmt(mcap)      : 'N/A', cls: '' },
    { l:'Beta (תנודתיות)', v: beta  != null ? fmtNum(beta,2) : 'N/A', cls: beta && beta>1.5?'warn':'' },
    { l:'שולי רווח גולמי', v: fmtPct(gm),   cls: colorClass(gm,  0.4, 0.2) },
    { l:'שולי רווח נקי',   v: fmtPct(pm),   cls: colorClass(pm,  0.15,0.05) },
    { l:'ROE',             v: fmtPct(roe),  cls: colorClass(roe, 0.15,0.05) },
    { l:'ROA',             v: fmtPct(roa),  cls: colorClass(roa, 0.08,0.02) },
    { l:'צמיחת הכנסות',   v: fmtPct(revGr),cls: colorClass(revGr,0.1,0) },
    { l:'צמיחת EPS (Q)',   v: fmtPct(epsGr),cls: colorClass(epsGr,0.1,0) },
    { l:'חוב / הון (D/E)', v: d2e   != null ? fmtNum(d2e,0)  : 'N/A', cls: colorClass(d2e, 50, 150, true) },
    { l:'Current Ratio',   v: cr    != null ? fmtNum(cr,2)   : 'N/A', cls: colorClass(cr,  1.5, 1.0) },
    { l:'Free Cash Flow',  v: fcf   != null ? fmt(fcf)       : 'N/A', cls: fcf>0?'good':'bad' },
    { l:'יעד אנליסטים',   v: targetP ? curSym+fmtNum(targetP,2) : 'N/A',
      cls: upside!=null ? (upside>15?'good':upside<-5?'bad':'warn') : '' },
  ];

  document.getElementById('metric-grid').innerHTML = metrics.map(m => `
    <div class="metric-card">
      <div class="metric-label">${m.l}</div>
      <div class="metric-value ${m.cls}">${m.v}</div>
    </div>
  `).join('');
}

/* ── CHART ── */
function renderChart(priceData, isPositive) {
  const ctx = document.getElementById('price-chart').getContext('2d');
  if (chartInstance) { chartInstance.destroy(); chartInstance = null; }

  const quotes  = priceData.indicators?.quote?.[0] || {};
  const closes  = quotes.close  || [];
  const times   = (priceData.timestamp || []).map(t =>
    new Date(t*1000).toLocaleDateString('he-IL', {month:'numeric',day:'numeric'})
  );

  // Filter out null data points
  const valid = closes.reduce((acc, c, i) => {
    if (c != null) acc.push(i);
    return acc;
  }, []);

  const labels   = valid.map(i => times[i]);
  const closeArr = valid.map(i => closes[i]);
  const maxClose = Math.max(...closeArr);
  const minClose = Math.min(...closeArr);
  const padding  = (maxClose - minClose) * 0.1;

  const color = isPositive ? '#22d87a' : '#f05c5c';

  const gradient = ctx.createLinearGradient(0, 0, 0, 250);
  gradient.addColorStop(0, isPositive ? 'rgba(34,216,122,0.2)' : 'rgba(240,92,92,0.2)');
  gradient.addColorStop(1, 'rgba(0,0,0,0)');

  chartInstance = new Chart(ctx, {
    type: 'line',
    data: {
      labels,
      datasets: [{
        label: 'מחיר סגירה',
        data: closeArr,
        borderColor: color,
        borderWidth: 2,
        pointRadius: 0,
        pointHoverRadius: 5,
        pointHoverBackgroundColor: color,
        fill: true,
        backgroundColor: gradient,
        tension: 0.35,
      }]
    },
    options: {
      responsive: true,
      maintainAspectRatio: false,
      interaction: { mode: 'index', intersect: false },
      plugins: {
        legend: { display: false },
        tooltip: {
          backgroundColor: '#0d1017',
          borderColor: 'rgba(255,255,255,0.08)',
          borderWidth: 1,
          titleColor: '#8b92a8',
          bodyColor: '#eef0f6',
          bodyFont: { family: "'DM Mono', monospace" },
          callbacks: {
            label: ctx => ' ' + ctx.parsed.y.toFixed(2)
          }
        }
      },
      scales: {
        x: {
          grid: { color: 'rgba(255,255,255,0.03)' },
          ticks: { color: '#454d61', font: { size: 10, family: "'DM Mono', monospace" }, maxTicksLimit: 8 }
        },
        y: {
          grid: { color: 'rgba(255,255,255,0.04)' },
          ticks: { color: '#8b92a8', font: { size: 10, family: "'DM Mono', monospace" },
                   callback: v => v.toFixed(0) },
          min: minClose - padding,
          max: maxClose + padding,
        }
      }
    }
  });
}

  const detail = fund.summaryDetail        || {};
  const profile= fund.assetProfile         || {};
  const meta   = price.meta;

  const cur       = meta.regularMarketPrice;
  const prev      = meta.chartPreviousClose || cur;
  const pctChange = prev ? ((cur-prev)/prev*100).toFixed(2) : 'N/A';
  const targetP   = fin.targetMeanPrice?.raw;
  const upside    = (targetP && cur) ? ((targetP - cur)/cur*100).toFixed(1) : 'N/A';

  const payload = {
    model: 'claude-sonnet-4-20250514',
    max_tokens: 1000,
    messages: [{
      role: 'user',
      content: `אתה אנליסט מניות מקצועי בכיר. נתח את המניה הבאה בשיטת ה-Fundamental Analysis ותן המלצה ברורה.

**מחיר נוכחי:** ${cur?.toFixed(2)} ${meta.currency}

**שווי שוק:** ${stats.marketCap?.fmt || 'N/A'}

**מכפילים:**
- P/E Trailing: ${stats.trailingPE?.fmt || 'N/A'}
- P/E Forward: ${stats.forwardPE?.fmt || 'N/A'}

  document.getElementById('r-verdict').textContent = '...';

    renderVerdict(fallback);
  }

async function typeText(el, text) {
  el.innerHTML = '';
  const cursor = document.createElement('span');
  cursor.className = 'ai-cursor';
  el.appendChild(cursor);

  for (let i = 0; i < text.length; i++) {
    const node = document.createTextNode(text[i]);
    el.insertBefore(node, cursor);
    if (i % 3 === 0) await new Promise(r => setTimeout(r, 12));
  }
  cursor.remove();
}

function renderVerdict(v) {
  const score = Math.min(100, Math.max(0, parseInt(v.score) || 50));
  const verdict = v.verdict || 'HOLD';
  const confidence = v.confidence || 'בינונית';

  document.getElementById('r-score').textContent      = score;
  document.getElementById('r-confidence').textContent = confidence;

  const vEl = document.getElementById('r-verdict');
  vEl.textContent = verdict;
  vEl.style.color = verdict === 'BUY' ? 'var(--green)' :
                    verdict === 'SELL'? 'var(--red)' : 'var(--yellow)';

  // Animate ring
  const arc = document.getElementById('score-arc');
  const scoreColor = score >= 65 ? 'var(--green)' :
                     score >= 40 ? 'var(--yellow)' : 'var(--red)';
  arc.style.stroke = scoreColor;
  arc.style.strokeDashoffset = 188.5 - (188.5 * score / 100);
  document.getElementById('r-score').style.color = scoreColor;
}

function calcFallbackScore(fund) {
  const fin   = fund.financialData        || {};
  const stats = fund.defaultKeyStatistics || {};
  let score = 50;
  if ((fin.profitMargins?.raw || 0) > 0.15) score += 15;
  if ((fin.revenueGrowth?.raw || 0) > 0.1)  score += 10;
  if ((stats.trailingPE?.raw || 999) < 25)   score += 10;
  if ((fin.returnOnEquity?.raw || 0) > 0.15) score += 10;
  if ((fin.freeCashflow?.raw || 0) > 0)      score += 5;
  score = Math.min(95, score);
  return {
    score,
    verdict: score >= 65 ? 'BUY' : score >= 40 ? 'HOLD' : 'SELL',
    confidence: 'נמוכה'
  };
}
