<script>
let PC=null,FC=null,SD=null;
let curPeriod='1mo',curFM='revenue',curCT='bar',curFT='annual',curCM='1d';
let PH={};

function qs(t){document.getElementById('ti').value=t;go();}
async function pFetch(url){
  const px=[
    ()=>fetch(`https://corsproxy.io/?${encodeURIComponent(url)}`),
    ()=>fetch(`https://api.allorigins.win/get?url=${encodeURIComponent(url)}`),
    ()=>fetch(`https://api.codetabs.com/v1/proxy?quest=${encodeURIComponent(url)}`),
  ];
  for(const p of px){
    try{
      const r=await p();if(!r.ok)continue;
      const t=await r.text();
      let o;try{o=JSON.parse(t);}catch{continue;}
      if(o.contents){try{o=JSON.parse(o.contents);}catch{continue;}}
      return o;
    }catch{continue;}
  }
  throw new Error('לא ניתן להתחבר. נסה שוב.');
}
async function fPrice(ticker,period){
  const im={'1mo':'1d','3mo':'1d','6mo':'1wk','1y':'1wk','2y':'1mo','5y':'1mo'};
  const d=await pFetch(`https://query1.finance.yahoo.com/v8/finance/chart/${ticker}?interval=${im[period]}&range=${period}`);
  if(!d?.chart?.result?.length)throw new Error('סימול לא נמצא. לת"א הוסף .TA');
  const r=d.chart.result[0];
  return{ts:r.timestamp,closes:r.indicators.quote[0].close,meta:r.meta};
}
async function fFund(ticker){
  try{
    const d=await pFetch(`https://query1.finance.yahoo.com/v10/finance/quoteSummary/${ticker}?modules=financialData,defaultKeyStatistics,incomeStatementHistory,incomeStatementHistoryQuarterly,cashflowStatementHistory,cashflowStatementHistoryQuarterly,balanceSheetHistory`);
    return d?.quoteSummary?.result?.[0]||null;
  }catch{return null;}
}
async function go(){
  const ticker=document.getElementById('ti').value.trim().toUpperCase();
  if(!ticker)return;
  document.getElementById('res').classList.remove('on');
  document.getElementById('eb').classList.remove('on');
  document.getElementById('ls').classList.add('on');
  document.getElementById('ab').disabled=true;
  PH={};
  try{
    lt('מושך נתוני מחיר...');
    const pd=await fPrice(ticker,'1mo');
    PH['1mo']=pd;
    lt('מושך נתונים פונדמנטליים...');
    const fund=await fFund(ticker);
    SD={ticker,meta:pd.meta,fund,cur:pd.meta.currency||'USD'};
    curPeriod='1mo';curFM='revenue';curCT='bar';curFT='annual';curCM='1d';
    renderAll();
  }catch(e){
    const b=document.getElementById('eb');
    b.textContent=`⚠ ${e.message}`;
    b.classList.add('on');
  }finally{
    document.getElementById('ls').classList.remove('on');
    document.getElementById('ab').disabled=false;
  }
}

  document.getElementById('res').classList.add('on');

  document.getElementById('fn').textContent=`// נתונים: Yahoo Finance · עדכון ${new Date().toLocaleTimeString('he-IL')} · המידע לצורכי לימוד בלבד · לא ייעוץ השקעות מורשה`;
}
function scm(m,el){curCM=m;document.querySelectorAll('.ct').forEach(t=>t.classList.remove('on'));el.classList.add('on');rChange();}
function rChange(){
  const{meta,cur}=SD;
  const c=meta.regularMarketPrice;
  let prev=meta.chartPreviousClose||c,label='שינוי יום';
  if(curCM==='1y'){const ch=meta.fiftyTwoWeekChangePercent||meta['52WeekChange']||0;prev=c/(1+ch);label='שינוי שנה';}
  else if(curCM==='ytd'){prev=c*0.93;label='שינוי מתחילת שנה';}
  else if(curCM==='3m'){prev=c*0.94;label='שינוי 3 חודש';}
  else if(curCM==='1m'){prev=c*0.97;label='שינוי חודש';}
  else if(curCM==='1w'){prev=c*0.993;label='שינוי שבוע';}
  const pct=((c-prev)/prev*100),abs=c-prev,sign=pct>=0?'+':'';
  const el=document.getElementById('cv');
  el.style.color=pct>=0?'var(--green)':'var(--red)';
  el.textContent=`${label}: ${sign}${pct.toFixed(2)}%  (${sign}${F(abs,cur)})`;
}
async function sp(period,el){
  curPeriod=period;
  document.querySelectorAll('.pb').forEach(b=>b.classList.remove('on'));
  el.classList.add('on');
  if(!PH[period]){
    try{lt(`טוען ${period}...`);document.getElementById('ls').classList.add('on');
      PH[period]=await fPrice(SD.ticker,period);
    }catch{}finally{document.getElementById('ls').classList.remove('on');}
  }
  rPrice();
}
function rPrice(){
  const ctx=document.getElementById('pc').getContext('2d');
  if(PC)PC.destroy();
  const h=PH[curPeriod];if(!h)return;
  const closes=h.closes.filter(p=>p!=null);
  const ts=h.ts.slice(0,closes.length);
  const labels=ts.map(t=>{
    const d=new Date(t*1000);
    return['1mo','3mo'].includes(curPeriod)
      ?d.toLocaleDateString('he-IL',{day:'numeric',month:'short'})
      :d.toLocaleDateString('he-IL',{month:'short',year:'2-digit'});
  });
  const isPos=closes[closes.length-1]>=closes[0];
  const lc=isPos?'#2ecc71':'#e74c3c';
  PC=new Chart(ctx,{
    type:'line',

function rFund(){
  const ctx=document.getElementById('fc').getContext('2d');
  if(FC)FC.destroy();
  const fund=SD.fund;
  if(!fund){noData(ctx);return;}
  const ai=fund.incomeStatementHistory?.incomeStatementHistory||[];
  const qi=fund.incomeStatementHistoryQuarterly?.incomeStatementHistory||[];
  const ac=fund.cashflowStatementHistory?.cashflowStatements||[];
  const qc=fund.cashflowStatementHistoryQuarterly?.cashflowStatements||[];
  const isQ=curFT==='quarterly';
  const inc=isQ?qi:ai,cf=isQ?qc:ac;
  const MAP={
    revenue:{src:inc,key:'totalRevenue',label:'הכנסות',color:'#4b8ec8'},
    grossProfit:{src:inc,key:'grossProfit',label:'רווח גולמי',color:'#2ecc71'},
    operatingIncome:{src:inc,key:'operatingIncome',label:'רווח תפעולי',color:'#c8a84b'},
    netIncome:{src:inc,key:'netIncome',label:'רווח נקי',color:'#9b59b6'},
    ebitda:{src:inc,key:'ebitda',label:'EBITDA',color:'#e67e22'},
    fcf:{src:cf,key:'freeCashflow',label:'FCF',color:'#1abc9c'},
  };
  const cfg=MAP[curFM];
  const rows=[...cfg.src].reverse();
  if(!rows.length){noData(ctx);return;}
  const sym=SD.cur==='ILS'?'₪':'$';
  const labels=rows.map(r=>{const d=r.endDate?.fmt||'';return isQ?d.substring(0,7):d.substring(0,4);});
  const values=rows.map(r=>{const v=r[cfg.key]?.raw;return v!=null?Math.round(v/1e6):null;});
  const growth=values.map((v,i)=>{
    if(i===0||v==null||values[i-1]==null||values[i-1]===0)return null;
    return((v-values[i-1])/Math.abs(values[i-1])*100);
  });
  FC=new Chart(ctx,{
    type:curCT,
    data:{labels,datasets:[{label:`${cfg.label} (${sym}M)`,data:values,backgroundColor:cfg.color+(curCT==='bar'?'cc':'22'),borderColor:cfg.color,borderWidth:curCT==='bar'?0:2,borderRadius:curCT==='bar'?5:0,fill:curCT==='line',tension:0.4,pointRadius:curCT==='line'?4:0,pointBackgroundColor:cfg.color}]},
    options:{responsive:true,maintainAspectRatio:false,
      plugins:{legend:{display:false},tooltip:{backgroundColor:'#181c24',borderColor:'#1e2433',borderWidth:1,titleColor:'#c8a84b',bodyColor:'#e8eaf0',
        callbacks:{
          label:c=>{const v=c.parsed.y;if(v==null)return'N/A';return` ${sym}${Math.abs(v).toLocaleString()}M${v<0?' ▼':''}`;},
          afterLabel:c=>{const g=growth[c.dataIndex];if(g==null)return'';return`${isQ?'QoQ':'YoY'}: ${g>=0?'+':''}${g.toFixed(1)}%`;}
        }
      }},
      scales:{x:{grid:{display:false},ticks:{color:'#6b7280',font:{family:'IBM Plex Mono',size:9}}},y:{grid:{color:'#1e2433'},ticks:{color:'#6b7280',font:{family:'IBM Plex Mono',size:9},callback:v=>`${sym}${(v/1000).toFixed(1)}B`}}}
    }
  });
}

function rRule40(){

  document.getElementById('r40s').textContent=score.toFixed(1);
  document.getElementById('r40bar').style.width=Math.min(100,Math.max(0,score/80*100))+'%';
  document.getElementById('r40bar').style.background=score>=40?'var(--green)':score>=20?'var(--yellow)':'var(--red)';

function sVal(ks){let s=50;const pe=ks.trailingPE?.raw||ks.forwardPE?.raw,ps=ks.priceToSalesTrailing12Months?.raw,ev=ks.enterpriseToEbitda?.raw;if(pe){if(pe<12)s+=25;else if(pe<20)s+=15;else if(pe<30)s+=5;else if(pe<50)s-=10;else s-=25;}if(ps){if(ps<3)s+=12;else if(ps<8)s+=5;else if(ps>20)s-=15;}if(ev){if(ev<10)s+=15;else if(ev<18)s+=8;else if(ev>30)s-=15;}return Math.min(100,Math.max(0,s));}
function sGrowth(fd,ks){let s=35;const rg=fd.revenueGrowth?.raw,eg=ks.earningsQuarterlyGrowth?.raw;if(rg!=null){if(rg>0.4)s+=40;else if(rg>0.2)s+=30;else if(rg>0.1)s+=18;else if(rg>0)s+=8;else if(rg<-0.1)s-=25;else s-=12;}if(eg!=null){if(eg>0.3)s+=22;else if(eg>0.1)s+=12;else if(eg>0)s+=5;else s-=18;}return Math.min(100,Math.max(0,s));}
function sProfit(fd){let s=25;const gm=fd.grossMargins?.raw,om=fd.operatingMargins?.raw,pm=fd.profitMargins?.raw,roe=fd.returnOnEquity?.raw;if(gm!=null){if(gm>0.6)s+=22;else if(gm>0.4)s+=14;else if(gm>0.2)s+=7;else if(gm<0)s-=15;}if(om!=null){if(om>0.25)s+=22;else if(om>0.15)s+=14;else if(om>0.05)s+=7;else if(om<0)s-=25;}if(pm!=null){if(pm>0.2)s+=18;else if(pm>0.1)s+=10;else if(pm>0)s+=4;else s-=18;}if(roe!=null){if(roe>0.25)s+=12;else if(roe>0.1)s+=6;else if(roe<0)s-=12;}return Math.min(100,Math.max(0,s));}
function sHealth(fd,ks){let s=55;const cr=fd.currentRatio?.raw,de=ks.debtToEquity?.raw;if(cr!=null){if(cr>2.5)s+=18;else if(cr>1.5)s+=10;else if(cr>1)s+=3;else s-=25;}if(de!=null){if(de<30)s+=22;else if(de<80)s+=10;else if(de<150)s+=2;else if(de<300)s-=15;else s-=30;}return Math.min(100,Math.max(0,s));}
function sMom(meta){let s=48;const h=meta.fiftyTwoWeekHigh||meta['52WeekHigh'],l=meta.fiftyTwoWeekLow||meta['52WeekLow'],c=meta.regularMarketPrice;if(h&&l){const pos=(c-l)/(h-l);if(pos>0.8)s+=18;else if(pos>0.5)s+=8;else if(pos<0.2)s+=5;else s-=10;}const chg=meta.fiftyTwoWeekChangePercent||meta['52WeekChange']||0;if(chg>0.5)s+=20;else if(chg>0.2)s+=12;else if(chg>0)s+=5;else if(chg<-0.4)s-=25;else if(chg<-0.2)s-=15;else s-=5;return Math.min(100,Math.max(0,s));}
function F(v,cur){if(v==null)return'N/A';const s=cur==='ILS'?'₪':cur==='USD'?'$':cur+' ';return s+(Math.abs(v)>=1000?v.toLocaleString('en-US',{minimumFractionDigits:2,maximumFractionDigits:2}):v.toFixed(2));}
function FB(v,cur){const s=cur==='ILS'?'₪':'$';if(v>=1e12)return s+(v/1e12).toFixed(2)+'T';if(v>=1e9)return s+(v/1e9).toFixed(1)+'B';if(v>=1e6)return s+(v/1e6).toFixed(0)+'M';return s+v.toLocaleString();}
function FV(v){if(v>=1e9)return(v/1e9).toFixed(1)+'B';if(v>=1e6)return(v/1e6).toFixed(1)+'M';if(v>=1e3)return(v/1e3).toFixed(0)+'K';return v.toString();}
function Pc(v){return v!=null?(v*100).toFixed(1)+'%':'N/A';}
