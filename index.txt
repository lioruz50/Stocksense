<script>
const DEFAULT_API_KEY = 'demo';
let chartInstance = null;
let FMP_KEY = localStorage.getItem('fmp_key') || DEFAULT_API_KEY;

window.onload = () => {
  if (FMP_KEY && FMP_KEY !== 'demo') {
    document.getElementById('api-notice').style.display = 'none';
  }
};

function saveKey() {
  const k = document.getElementById('api-key-input').value.trim();
  if (!k) return;
  FMP_KEY = k;
  localStorage.setItem('fmp_key', k);
  document.getElementById('api-notice').style.display = 'none';
  showInfo('✓ API Key נשמר בהצלחה');
}

function showInfo(msg) {
  const e = document.getElementById('error-box');
  e.style.color = 'var(--green)';
  e.style.borderColor = 'rgba(34,216,122,0.3)';
  e.style.background = 'rgba(34,216,122,0.05)';
  e.innerHTML = msg;
  e.classList.add('on');
  setTimeout(() => e.classList.remove('on'), 3000);
}

function qs(t) { document.getElementById('ticker-input').value = t; go(); }

function setState(s) {
  document.getElementById('loading-box').classList.toggle('on', s === 'loading');
  document.getElementById('result').classList.toggle('on', s === 'result');
  if (s !== 'error') {
    const eb = document.getElementById('error-box');
    eb.classList.remove('on');
    eb.style.color = '';
    eb.style.borderColor = '';
    eb.style.background = '';
  }
  document.getElementById('analyze-btn').disabled = s === 'loading';
}

function showError(msg) {
  const e = document.getElementById('error-box');
  e.style.color = '';
  e.style.borderColor = '';
  e.style.background = '';
  e.innerHTML = '⚠ ' + msg;
  e.classList.add('on');
  setState('error');
}

async function fmpFetch(path) {
  const sep = path.includes('?') ? '&' : '?';
  const url = `https://financialmodelingprep.com/api${path}${sep}apikey=${FMP_KEY}`;
  const resp = await fetch(url);
  if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
  return resp.json();
}

async function go() {
  const raw = document.getElementById('ticker-input').value.trim().toUpperCase().replace(/[^A-Z.^-]/g, '');
  if (!raw) return;

  if (!FMP_KEY || FMP_KEY === 'demo') {
    showError('יש להכניס API Key תחילה. ההרשמה ב-financialmodelingprep.com חינמית.');
    return;
  }

  setState('loading');

  try {
    const [histData, profileData, ratiosData, incomeData] = await Promise.all([
      fmpFetch(`/v3/historical-price-full/${raw}?timeseries=30`),
      fmpFetch(`/v3/profile/${raw}`),
      fmpFetch(`/v3/ratios-ttm/${raw}`),
      fmpFetch(`/v3/income-statement/${raw}?limit=2`)
    ]);

    if (!profileData?.[0]) throw new Error(`לא נמצא סימול "${raw}". בדוק שזה סימול NYSE/NASDAQ תקין.`);

    renderAll(raw, histData, profileData[0], ratiosData?.[0] || {}, incomeData || []);
    setState('result');

  } catch (e) {
    console.error(e);
    if (e.message.includes('401') || e.message.includes('403')) {
      showError('API Key שגוי או פג תוקף. בדוק את ה-Key שלך.');
    } else if (e.message.includes('429')) {
      showError('חריגת מכסה יומית (250 בקשות ביום בחשבון חינמי). נסה מחר.');
    } else {
      showError(e.message || 'שגיאה לא צפויה. נסה שוב.');
    }
  }
}

function renderAll(ticker, hist, profile, ratios, income) {
  document.getElementById('r-name').textContent   = profile.companyName || ticker;
  document.getElementById('r-ticker').textContent = `${ticker} · ${profile.exchangeShortName || ''} · ${profile.currency || 'USD'}`;
  document.getElementById('r-for').textContent    = 'מנותח: ' + ticker;

  const price  = profile.price || 0;
  const change = profile.changes || 0;
  const pct    = price ? (change / (price - change) * 100) : 0;
  const curSym = profile.currency === 'ILS' ? '₪' : profile.currency === 'EUR' ? '€' : '$';

  document.getElementById('r-price').textContent = curSym + price.toFixed(2);
  const chEl = document.getElementById('r-change');
  chEl.innerHTML = `<span style="color:${change >= 0 ? 'var(--green)' : 'var(--red)'}">
    ${change >= 0 ? '▲' : '▼'} ${Math.abs(pct).toFixed(2)}% (${change >= 0 ? '+' : ''}${change.toFixed(2)})
  </span>`;

  renderChart(hist, change >= 0);

  const pe  = ratios.peRatioTTM;
  const pb  = ratios.priceToBookRatioTTM;
  const ps  = ratios.priceToSalesRatioTTM;
  const roe = ratios.returnOnEquityTTM;
  const roa = ratios.returnOnAssetsTTM;
  const pm  = ratios.netProfitMarginTTM;
  const gm  = ratios.grossProfitMarginTTM;
  const d2e = ratios.debtEquityRatioTTM;
  const cr  = ratios.currentRatioTTM;
  const beta   = profile.beta;
  const mcap   = profile.mktCap;
  const target = profile.dcf;
  const upside = (target && price) ? ((target - price) / price * 100) : null;

  let revGr = null;
  if (income.length >= 2) {
    const r1 = income[0]?.revenue;
    const r2 = income[1]?.revenue;
    if (r1 && r2 && r2 !== 0) revGr = (r1 - r2) / Math.abs(r2);
  }

  function f(n, d = 2)  { return n == null || isNaN(n) ? 'N/A' : Number(n).toFixed(d); }
  function fp(n)         { return n == null || isNaN(n) ? 'N/A' : (n * 100).toFixed(1) + '%'; }
  function fm(n) {
    if (n == null || isNaN(n)) return 'N/A';
    if (Math.abs(n) >= 1e12) return (n / 1e12).toFixed(2) + 'T';
    if (Math.abs(n) >= 1e9)  return (n / 1e9).toFixed(2) + 'B';
    if (Math.abs(n) >= 1e6)  return (n / 1e6).toFixed(2) + 'M';
    return n.toFixed(2);
  }
  function cc(v, g, b, inv = false) {
    if (v == null || isNaN(v)) return '';
    return (inv ? v < g : v > g) ? 'good' : (inv ? v > b : v < b) ? 'bad' : 'warn';
  }

  const metrics = [
    { l: 'P/E Ratio',         v: f(pe, 1),   cls: cc(pe, 15, 35, true) },
    { l: 'P/B (Price/Book)',  v: f(pb, 2),   cls: cc(pb, 1, 6, true) },
    { l: 'P/S (Price/Sales)', v: f(ps, 2),   cls: '' },
    { l: 'שווי שוק',          v: fm(mcap),   cls: '' },
    { l: 'Beta',              v: f(beta, 2), cls: beta && beta > 1.5 ? 'warn' : '' },
    { l: 'שולי רווח גולמי',   v: fp(gm),     cls: cc(gm, 0.4, 0.2) },
    { l: 'שולי רווח נקי',     v: fp(pm),     cls: cc(pm, 0.15, 0.05) },
    { l: 'ROE',               v: fp(roe),    cls: cc(roe, 0.15, 0.05) },
    { l: 'ROA',               v: fp(roa),    cls: cc(roa, 0.08, 0.02) },
    { l: 'צמיחת הכנסות',     v: fp(revGr),  cls: cc(revGr, 0.1, 0) },
    { l: 'חוב / הון (D/E)',   v: f(d2e, 2),  cls: cc(d2e, 0.5, 1.5, true) },
    { l: 'Current Ratio',     v: f(cr, 2),   cls: cc(cr, 1.5, 1.0) },
    { l: 'DCF שווי הוגן',     v: target ? curSym + f(target, 2) : 'N/A', cls: upside != null ? (upside > 15 ? 'good' : upside < -5 ? 'bad' : 'warn') : '' },
    { l: 'פוטנציאל DCF',      v: upside != null ? (upside > 0 ? '+' : '') + upside.toFixed(1) + '%' : 'N/A', cls: upside != null ? (upside > 15 ? 'good' : upside < -5 ? 'bad' : 'warn') : '' },
    { l: 'סקטור',             v: profile.sector   || 'N/A', cls: '' },
    { l: 'ענף',               v: profile.industry || 'N/A', cls: '' },
  ];

  document.getElementById('metric-grid').innerHTML = metrics.map(m => `
    <div class="metric-card">
      <div class="metric-label">${m.l}</div>
      <div class="metric-value ${m.cls}" title="${m.v}">${m.v.length > 14 ? m.v.substring(0, 13) + '…' : m.v}</div>
    </div>
  `).join('');

  renderVerdict(ratios, profile, upside, revGr, curSym);
}

function renderChart(hist, isPos) {
  const ctx = document.getElementById('price-chart').getContext('2d');
  if (chartInstance) { chartInstance.destroy(); chartInstance = null; }

  const raw = (hist?.historical || []).slice().reverse().slice(-30);
  if (!raw.length) return;

  const labels = raw.map(d => d.date.slice(5));
  const closes = raw.map(d => d.close);
  const minC   = Math.min(...closes);
  const maxC   = Math.max(...closes);
  const pad    = (maxC - minC) * 0.1 || 1;

  const color = isPos ? '#22d87a' : '#f05c5c';
  const grad  = ctx.createLinearGradient(0, 0, 0, 250);
  grad.addColorStop(0, isPos ? 'rgba(34,216,122,0.2)' : 'rgba(240,92,92,0.2)');
  grad.addColorStop(1, 'rgba(0,0,0,0)');

  chartInstance = new Chart(ctx, {
    type: 'line',
    data: {
      labels,
      datasets: [{
        data: closes, borderColor: color, borderWidth: 2,
        pointRadius: 0, pointHoverRadius: 5, pointHoverBackgroundColor: color,
        fill: true, backgroundColor: grad, tension: 0.35
      }]
    },
    options: {
      responsive: true, maintainAspectRatio: false,
      interaction: { mode: 'index', intersect: false },
      plugins: {
        legend: { display: false },
        tooltip: {
          backgroundColor: '#0d1017', borderColor: 'rgba(255,255,255,0.08)',
          borderWidth: 1, titleColor: '#8b92a8', bodyColor: '#eef0f6',
          bodyFont: { family: "'DM Mono',monospace" },
          callbacks: { label: c => ' ' + c.parsed.y.toFixed(2) }
        }
      },
      scales: {
        x: { grid: { color: 'rgba(255,255,255,0.03)' }, ticks: { color: '#454d61', font: { size: 10 }, maxTicksLimit: 8 } },
        y: { grid: { color: 'rgba(255,255,255,0.04)' }, min: minC - pad, max: maxC + pad,
             ticks: { color: '#8b92a8', font: { size: 10 }, callback: v => v.toFixed(0) } }
      }
    }
  });
}

function renderVerdict(ratios, profile, upside, revGr) {
  let score = 50;
  const pm  = ratios.netProfitMarginTTM;
  const roe = ratios.returnOnEquityTTM;
  const cr  = ratios.currentRatioTTM;
  const d2e = ratios.debtEquityRatioTTM;
  const pe  = ratios.peRatioTTM;

  if (pm  != null && pm  > 0.15) score += 10;
  if (pm  != null && pm  > 0.25) score += 5;
  if (roe != null && roe > 0.15) score += 10;
  if (revGr != null && revGr > 0.1) score += 10;
  if (revGr != null && revGr > 0.2) score += 5;
  if (cr  != null && cr  > 1.5)  score += 8;
  if (d2e != null && d2e < 0.5)  score += 7;
  if (pe  != null && pe  > 0 && pe < 20) score += 8;
  if (upside != null && upside > 15)  score += 10;
  if (upside != null && upside < -10) score -= 10;

  score = Math.min(98, Math.max(5, score));

  const verdict = score >= 68 ? 'BUY' : score >= 42 ? 'HOLD' : 'SELL';
  const color   = score >= 68 ? 'var(--green)' : score >= 42 ? 'var(--yellow)' : 'var(--red)';

  document.getElementById('r-score').textContent  = score;
  document.getElementById('r-score').style.color  = color;
  const vEl = document.getElementById('r-verdict');
  vEl.textContent = verdict;
  vEl.style.color = color;

  const uEl = document.getElementById('r-upside');
  if (upside != null) {
    uEl.textContent = (upside > 0 ? '+' : '') + upside.toFixed(1) + '%';
    uEl.style.color = upside > 15 ? 'var(--green)' : upside < -5 ? 'var(--red)' : 'var(--yellow)';
  } else {
    uEl.textContent = 'N/A';
  }

  const arc = document.getElementById('score-arc');
  arc.style.stroke = color;
  arc.style.strokeDashoffset = 188.5 - (188.5 * score / 100);

  const lines = [];
  if (pe != null && pe > 0) {
    lines.push(pe < 15
      ? `מכפיל הרווח (P/E=${pe.toFixed(1)}) נמוך יחסית לשוק – עשוי להצביע על תמחור נוח.`
      : pe < 30
      ? `מכפיל הרווח (P/E=${pe.toFixed(1)}) נמצא בטווח הסביר לשוק.`
      : `מכפיל הרווח (P/E=${pe.toFixed(1)}) גבוה – השוק מתמחר ציפיות צמיחה משמעותיות.`);
  }
  if (pm != null) {
    lines.push(pm > 0.2
      ? `שולי הרווח הנקי (${(pm * 100).toFixed(1)}%) גבוהים ומעידים על יעילות תפעולית מצוינת.`
      : pm > 0.05
      ? `שולי הרווח הנקי (${(pm * 100).toFixed(1)}%) סבירים.`
      : `שולי הרווח הנקי (${(pm * 100).toFixed(1)}%) נמוכים – יש לבחון את מבנה העלויות.`);
  }
  if (revGr != null) {
    lines.push(revGr > 0.15
      ? `צמיחת ההכנסות (${(revGr * 100).toFixed(1)}%) חזקה – החברה מרחיבה את פעילותה.`
      : revGr > 0
      ? `צמיחת הכנסות מתונה (${(revGr * 100).toFixed(1)}%).`
      : `ירידה בהכנסות (${(revGr * 100).toFixed(1)}%) – גורם לניטור.`);
  }
  if (upside != null) {
    lines.push(upside > 20
      ? `לפי מודל ה-DCF, המניה נסחרת בדיסקאונט משמעותי של ${upside.toFixed(1)}% מהשווי ההוגן.`
      : upside > 0
      ? `מודל DCF מצביע על פוטנציאל עלייה מתון של ${upside.toFixed(1)}%.`
      : `לפי DCF, המניה נסחרת מעל השווי ההוגן ב-${Math.abs(upside).toFixed(1)}%.`);
  }
  lines.push(verdict === 'BUY'
    ? `סיכום: יסודות חזקים תומכים בעמדת השקעה חיובית.`
    : verdict === 'SELL'
    ? `סיכום: הנתונים מצביעים על זהירות – שקול צמצום חשיפה.`
    : `סיכום: תמונה מעורבת – מומלץ להמתין לנתונים נוספים לפני פעולה.`);

  document.getElementById('r-analysis').textContent = lines.join('\n\n');
}
